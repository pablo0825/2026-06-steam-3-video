# Ch4 Page 4 PPU (S10-S12) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the 36-second `Ch4-Page4-PPU` Remotion composition for S10-S12 of 第 4 集「美術整合」, explaining Unity `unit`, image `px`, and `PPU（Pixels Per Unit）`.

**Architecture:** One page-level React component (`Ch4Page4PPU`) drives three sequential shots from a single 1080-frame timeline. S10 presents the `unit` / `px` conversion problem with large colored text, aligned two-way draw-on arrows, a question mark, and a longer final hold; S11 answers with a minimal PPU definition screen; S12 uses two comparison cards with overlaid `1 unit` and `128 px` squares to show why `PPU=128` makes a 128 px image equal exactly 1 unit. The page uses only inline SVG, divs, and text; no new image assets.

**Tech Stack:** React, TypeScript, Remotion (`AbsoluteFill`, `Easing`, `interpolate`, `spring`, `useCurrentFrame`, `useVideoConfig`), existing theme tokens from `src/theme/colors.ts`, inline SVG for arrows and simple measurement marks.

```tsx
import React from "react";
import {
  AbsoluteFill,
  Easing,
  interpolate,
  spring,
  useCurrentFrame,
  useVideoConfig,
} from "remotion";
import {
  BLUE,
  SUBTLE,
  TEXT_DARK,
  WHITE,
  YELLOW,
} from "../../theme/colors";

const FONT = '"Noto Sans TC", "Microsoft JhengHei", "PingFang TC", sans-serif';
const clamp = { extrapolateLeft: "clamp", extrapolateRight: "clamp" } as const;
const ease = { ...clamp, easing: Easing.bezier(0.16, 1, 0.3, 1) };

// S10: 0-240, S11: 240-630, S12: 630-1080
const S10_OUT = [216, 240] as const;
const S11_IN = [240, 268] as const;
const S11_OUT = [600, 630] as const;
const S12_IN = [630, 660] as const;

type UnitLabelProps = {
  label: string;
  sublabel: string;
  x: number;
  frame: number;
  fps: number;
  delay: number;
};

const UnitLabel: React.FC<UnitLabelProps> = ({
  label,
  sublabel,
  x,
  frame,
  fps,
  delay,
}) => {
  const enter = spring({
    frame: frame - delay,
    fps,
    config: { damping: 15, stiffness: 120 },
  });
  return (
    <div
      style={{
        position: "absolute",
        left: x,
        top: 520,
        transform: `translate(-50%, -50%) scale(${interpolate(enter, [0, 1], [0.9, 1])})`,
        opacity: enter,
        textAlign: "center",
      }}
    >
      <div
        style={{
          fontSize: 118,
          fontWeight: 900,
          letterSpacing: 1,
          color: TEXT_DARK,
          lineHeight: 1,
        }}
      >
        {label}
      </div>
      <div
        style={{
          marginTop: 22,
          fontSize: 34,
          fontWeight: 750,
          color: SUBTLE,
          letterSpacing: 2,
        }}
      >
        {sublabel}
      </div>
    </div>
  );
};

type DrawArrowProps = {
  x1: number;
  x2: number;
  y: number;
  progress: number;
  direction: "right" | "left";
};

const DrawArrow: React.FC<DrawArrowProps> = ({
  x1,
  x2,
  y,
  progress,
  direction,
}) => {
  const headX = direction === "right" ? x2 : x2;
  const rotation = direction === "right" ? 0 : 180;
  return (
    <g>
      <line
        x1={x1}
        y1={y}
        x2={x2}
        y2={y}
        stroke={BLUE}
        strokeWidth={7}
        strokeLinecap="round"
        pathLength={1}
        strokeDasharray={1}
        strokeDashoffset={1 - progress}
      />
      <g opacity={progress} transform={`translate(${headX} ${y}) rotate(${rotation})`}>
        <path d="M0 0 L-24 -14 L-24 14 Z" fill={BLUE} />
      </g>
    </g>
  );
};

export const Ch4Page4PPU: React.FC = () => {
  const frame = useCurrentFrame();
  const { fps } = useVideoConfig();

  const s10Opacity = interpolate(frame, S10_OUT, [1, 0], clamp);
  const titleIn = spring({
    frame,
    fps,
    config: { damping: 15, stiffness: 120 },
  });
  const arrowToPx = interpolate(frame, [66, 104], [0, 1], ease);
  const arrowToUnit = interpolate(frame, [92, 130], [0, 1], ease);
  const qIn = spring({
    frame: frame - 132,
    fps,
    config: { damping: 12, stiffness: 150 },
  });

  return (
    <AbsoluteFill style={{ backgroundColor: WHITE, fontFamily: FONT }}>
      {frame < 182 && (
        <AbsoluteFill style={{ opacity: s10Opacity }}>
          <div
            style={{
              position: "absolute",
              left: 960,
              top: 170,
              transform: `translateX(-50%) translateY(${interpolate(titleIn, [0, 1], [28, 0])}px)`,
              opacity: titleIn,
              fontSize: 64,
              fontWeight: 900,
              letterSpacing: 4,
              color: TEXT_DARK,
              whiteSpace: "nowrap",
            }}
          >
            素材大小的基礎單位
          </div>

          <UnitLabel label="unit" sublabel="Unity 單位" x={520} frame={frame} fps={fps} delay={24} />
          <UnitLabel label="px" sublabel="美術圖單位" x={1400} frame={frame} fps={fps} delay={42} />

          <svg width="1920" height="1080" viewBox="0 0 1920 1080" style={{ position: "absolute", inset: 0 }}>
            <DrawArrow x1={760} x2={1160} y={470} progress={arrowToPx} direction="right" />
            <DrawArrow x1={1160} x2={760} y={590} progress={arrowToUnit} direction="left" />
          </svg>

          <div
            style={{
              position: "absolute",
              left: 960,
              top: 530,
              transform: `translate(-50%, -50%) scale(${interpolate(qIn, [0, 1], [0.65, 1])})`,
              opacity: qIn,
              fontSize: 88,
              fontWeight: 950,
              color: YELLOW,
              lineHeight: 1,
            }}
          >
            ?
          </div>
        </AbsoluteFill>
      )}

      {/* S11 layer added in Task 2 */}
      {/* S12 layer added in Task 3 */}
    </AbsoluteFill>
  );
};
```
